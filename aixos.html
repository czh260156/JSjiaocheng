Users Found:

' + users.map(user => `
  • ${user.name}
    • `).join(''); }).catch(function (error) { // 失败的情况 console.log(error, 'erererererererererer+'); }) }, 1000)) // 1秒的延迟，用户停止点击1秒后才发送请求 // post请求 // 即使用户疯狂点击提交按钮，我们也给他进行限制，也是最多让他5秒内只处理一次点击，所以我们用节流函数 document.getElementById("postButton").addEventListener('click', throttle(function () { const title = document.getElementById('titleInput').value || '默认标题'; const body = document.getElementById('bodyInput').value || '默认内容'; const postData = { title: title, body: body, userId: 1 }; axios.post('https://jsonplaceholder.typicode.com/posts', postData).then(function (response) { // 处理成功情况 console.log('post成功++++++++++++', response.data); const postResultDiv = document.getElementById('postResult'); postResultDiv.innerHTML = `

    Post结果

    响应状态: ${response.status}

    响应状态: ${response.data.id}

    响应状态: ${response.data.title}

    响应状态: ${response.data.body}

    `; }).catch(function (error) { // 处理错误的情况 console.error('post失败++++++++++++++++', error); document.getElementById('postResult').innerHTML = `

    提交数据时出错

    错误信息: ${error}

    `; }) }, 2000))
